import moment, { Moment } from "moment-mini";

/**
 * The format string for dates in the 'YYYY-MM-DD' format.
 */
const DATE_FORMAT = "YYYY-MM-DD";

/**
 * Returns the date range for the last X days from a given base date.
 * @param days The number of days to go back.
 * @param baseDate The base date to start from. If not provided, the current date will be used.
 * @returns An object containing the start and end dates of the date range.
 */
export function getDateRangeLastXDays(days: number, baseDate?: string) {
    const base = moment.utc(baseDate);
    return {
        startDate: base.clone().utc().subtract(days, "days").startOf("day")
            .format(DATE_FORMAT),
        endDate: base.clone().utc().endOf("day").format(DATE_FORMAT),
    };
}

/**
 * Gets a range which includes the last X full calendar months along with all days in current month up to today.
 * So if X=3 and today = 15th May, will return 1st Feb to 15th May.
 * @param months
 * @param baseDate
 */
export function getDateRangeLastXFullMonths(months: number, baseDate?: string) {
    const base = moment.utc(baseDate);
    return {
        startDate: base.clone().utc().startOf("month").subtract(months, "months")
            .format(DATE_FORMAT),
        endDate: base.clone().utc().endOf("day").format(DATE_FORMAT),
    };
}

/**
 * Represents a date range with a start and end date.
 */
export interface DateRange {
    /**
     * The start date of the range.
     */
    start: Moment;

    /**
     * The end date of the range.
     */
    end: Moment;
}

/**
 * Checks if two date ranges overlap.
 * @param first - The first date range.
 * @param second - The second date range.
 * @returns True if the date ranges overlap, false otherwise.
 */
export function twoRangesOverlapped(first: DateRange, second: DateRange) {
    return first.start.isBefore(second.end) && first.end.isAfter(second.start);
}

/**
 * Checks if one date range is completely inside another date range.
 * @param given - The date range to check if it is inside the target date range.
 * @param target - The target date range to check against.
 * @returns True if the given date range is inside the target date range, false otherwise.
 */
export function oneRangeInsideAnother(given: DateRange, target: DateRange) {
    return given.start.isSameOrAfter(target.start) && given.end.isSameOrBefore(target.end);
}

/**
 * Converts a day and slot time into a Moment.js date object.
 * @param day - The day in string format.
 * @param slot - The slot time in string format.
 * @returns A Moment.js date object representing the specified day and slot time.
 */
export function getMomentDateFromDaySlotTime(day: string, slot: string) {
    return moment.utc(day)
        .hours(Number.parseInt(slot.slice(0, 2), 10))
        .minutes(Number.parseInt(slot.slice(2, 4), 10));
}
