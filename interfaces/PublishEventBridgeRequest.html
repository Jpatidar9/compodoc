import { EventBridgeClient, PutEventsCommand } from "@aws-sdk/client-eventbridge";
import { PublishCommand, SNSClient } from "@aws-sdk/client-sns";
import { EventBridgeEvent, EventBridgeSource } from "@common/types/types";

import { eventBridge } from "../config";
import log from "./log";

/**
 * The eventBridge object contains configuration settings for the event bridge.
 */
const {
    serviceBusName,
    defaultSource,
} = eventBridge;

/**
 * The EventBridge client used for publishing events.
 */
const eb = new EventBridgeClient({ region: process.env.AWS_REGION || "us-east-1" });

/**
 * AWS Simple Notification Service (SNS) client.
 */
const sns = new SNSClient({ region: process.env.AWS_REGION || "us-east-1" });

/**
 * Publishes a message to an SNS topic.
 * @param {string} topicArn - The ARN of the SNS topic.
 * @param {string} eventName - The name of the event.
 * @param {any} message - The message to be published.
 * @throws Will throw an error if publishing fails.
 */
export async function publishSNS(topicArn: string, eventName: string, message: any) {
    const publishRequest = {
        Message: JSON.stringify(message || {}),
        MessageAttributes: {
            eventName: {
                DataType: "String",
                StringValue: eventName,
            },
        },
        TopicArn: topicArn,
    };
    try {
        log.debug("Publishing event to SNS topic...", { publishRequest });
        const command = new PublishCommand(publishRequest);
        const result = await sns.send(command);
        log.debug("Published event to SNS topic", {
            publishRequest,
            result,
        });
    } catch (error: any) {
        log.error("Error publishing event to SNS topic", { publishRequest }, error);
        throw error;
    }
}

/**
 * Represents a request to publish an event to EventBridge.
 */
interface PublishEventBridgeRequest {
    /**
     * The source of the event.
     */
    source?: EventBridgeSource;

    /**
     * The name of the event.
     */
    eventName: EventBridgeEvent;

    /**
     * The message payload of the event.
     */
    eventMessage: any;
}

/**
 * Publishes an event to EventBridge.
 * @param {PublishEventBridgeRequest} request - The request object containing event details.
 * @throws Will throw an error if publishing fails or if one or more events fail to publish.
 */
export async function publishEventBridge(
    { eventName, eventMessage, source = defaultSource }: PublishEventBridgeRequest,
) {
    const publishRequest = {
        Entries: [{
            EventBusName: serviceBusName,
            Source: source,
            DetailType: eventName,
            Detail: JSON.stringify(eventMessage),
        }],
    };
    let result;
    try {
        log.debug("Publishing event to EventBridge...", { publishRequest });
        const command = new PutEventsCommand(publishRequest);
        result = await eb.send(command);
    } catch (error: any) {
        log.error("Error publishing event to EventBridge", { publishRequest }, error);
        throw error;
    }
    if (result.FailedEntryCount) {
        log.error("Error publishing one or more events to EventBridge", {
            publishRequest,
            result,
        });
        throw new Error("Error publishing one or more events to EventBridge");
    }
    log.debug("Published event to EventBridge", {
        publishRequest,
        result,
    });
}
