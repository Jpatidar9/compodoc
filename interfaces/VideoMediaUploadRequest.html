import { DeleteObjectCommand, DeleteObjectsCommand, GetObjectCommand, GetObjectCommandInput, PutObjectCommand, PutObjectCommandInput, S3Client } from "@aws-sdk/client-s3";

    getFileSuffixForContentType,

/**
 * Creates a new S3 client instance.
 * @param {string} [options.region="us-east-1"] - The AWS region to use. Defaults to "us-east-1" if not provided.
 * @returns {S3Client} The S3 client instance.
 */
const s3 = new S3Client({ region: process.env.AWS_REGION || "us-east-1" });

/**
 * Represents a media upload request.
 * @interface
 */
export interface MediaUploadRequest {
    /**
     * The content type of the media being uploaded.
     * @type {string}
     */
    contentType: string;

    /**
     * The type of entity the media is being uploaded for.
     * @type {UploadEntityType}
     */
    uploadEntityType: UploadEntityType;

    /**
     * The ID of the entity the media is associated with.
     * @type {string}
     */
    entityId: string;

    /**
     * The name of the target S3 bucket where the media will be uploaded.
     * @type {string}
     */
    targetBucket: string;
}

/**
 * Represents a video media upload request, extending the MediaUploadRequest interface.
 * @interface
 */
export interface VideoMediaUploadRequest extends MediaUploadRequest {
    /**
     * The ID of the user associated with the video media upload request.
     * @type {string}
     */
    userId: string;

    /**
     * The ID of the user associated with the video media upload request.
     * @type {string}
     */
    mentorId: string;

    /**
     * Additional details related to the video media upload request.
     * @type {any}
     */
    extraDetails: any;
}

/**
 * Represents a media upload response.
 * @interface
 */
export interface MediaUploadResponse {
    /**
     * The ID of the uploaded media.
     * @type {string}
     */
    mediaId: string;

    /**
     * The URL used to PUT the media object to S3.
     * @type {string}
     */
    s3PutObjectUrl: string;

    /**
     * The URL of the uploaded media.
     * @type {string}
     */
    mediaUrl: string;

    /**
     * The S3 bucket key associated with the uploaded media.
     * @type {string}
     */
    mediaBucketKey: string;
}

/**
 * Represents input parameters for obtaining a media bucket key.
 * @interface
 */
interface GetMediaBucketKeyInput {
    /**
     * The type of entity for which the media is being uploaded.
     * @type {UploadEntityType}
     */
    uploadEntityType: UploadEntityType;
    
    /**
     * The ID of the media.
     * @type {string}
     */
    mediaId: string;

    /**
     * The file suffix of the media.
     * @type {string}
     */
    fileSuffix: string;
}

/**
 * Function to get media bucket key
 * @param {GetMediaBucketKeyInput} input - The input parameters for getting media bucket key
 * @returns {string} The media bucket key
 */
export const getMediaBucketKey = ({
    uploadEntityType,
    mediaId,
    fileSuffix,
}: GetMediaBucketKeyInput): string => `${s3Config.mediaPrefix}${uploadEntityType}/${mediaId}.${fileSuffix}`;

/**
 * Function to get CloudFront URL from bucket key
 * @param {string} bucketKey - The bucket key
 * @returns {string} The CloudFront URL
 */
export const getCloudFrontUrlFromBucketKey = (bucketKey: string): string => `https://${cloudfront.mediaDistributionDomainName}/${bucketKey}`;

/**
 * Async function to generate S3 signed URL for image upload
 * @param {MediaUploadRequest} request - The media upload request
 * @returns {Promise<MediaUploadResponse>} The media upload response
 */
export async function generateS3SignedUrlForImageUpload(
    request: MediaUploadRequest,
): Promise<MediaUploadResponse> {
    if (!isValidContentType(request.contentType)) {
        throw new StatusCodeError(400, `Invalid contentType for media: ${request.contentType}. Valid values are: ${getSupportedContentTypes().join(",")}`);
    }
    const mediaId = ulid();
    const mediaBucketKey = getMediaBucketKey(
        {
            uploadEntityType: request.uploadEntityType!,
            mediaId,
            fileSuffix: getFileSuffixForContentType(request.contentType)!,
        },
    );

    // Create a signed PutObjectRequest and return its URL to the client
    const s3Req: PutObjectCommandInput = {
        Bucket: request.targetBucket,
        Key: mediaBucketKey,
        ContentType: request.contentType,
        CacheControl: "max-age=31557600", // cache in CDN for 1 year
        Metadata: {
            mediaId,
            ...request,
        },
    };
    const command = new PutObjectCommand(s3Req);
    await s3.send(command);

    // Get a signed URL for the uploaded object (valid for 1 hour)
    const s3PutObjectUrl = await getSignedUrl(s3, command, { expiresIn: 3600 });

    const mediaUrl = getCloudFrontUrlFromBucketKey(s3Req.Key!);

    return {
        mediaId, s3PutObjectUrl, mediaUrl, mediaBucketKey,
    };
}

/**
 * Async function to generate S3 signed URL for image upload
 * @param {MediaUploadRequest} request - The media upload request
 * @returns {Promise<MediaUploadResponse>} The media upload response
 */
export async function generateS3SignedUrlForVideoUpload(
    request: VideoMediaUploadRequest,
): Promise<MediaUploadResponse> {
    if (!isValidContentType(request.contentType)) {
        throw new StatusCodeError(400, `Invalid contentType for media: ${request.contentType}. Valid values are: ${getSupportedContentTypes().join(",")}`);
    }
    const mediaId = ulid();

    const mediaBucketKey = getMediaBucketKey(
        {
            uploadEntityType: request.uploadEntityType!,
            mediaId,
            fileSuffix: getFileSuffixForContentType(request.contentType)!,
        },
    );

    // Create a signed PutObjectRequest and return its URL to the client
    const s3Req: PutObjectCommandInput = {
        Bucket: request.targetBucket,
        Key: mediaBucketKey,
        ContentType: request.contentType,
        CacheControl: "max-age=31557600", // cache in CDN for 1 year
        Metadata: {
            mediaId,
            ...request,
            extraDetails: JSON.stringify(request.extraDetails),
        },
    };

    const command = new PutObjectCommand(s3Req);
    await s3.send(command);

    // Get a signed URL for the uploaded object (valid for 1 hour)
    const s3PutObjectUrl = await getSignedUrl(s3, command, { expiresIn: 3600 });
    const mediaUrl = getCloudFrontUrlFromBucketKey(s3Req.Key!);
    return {
        mediaId, s3PutObjectUrl, mediaUrl, mediaBucketKey,
    };
}

/**
 * Async function to generate signed get URL
 * @param {string} key - The key
 * @param {number} durationHrs - The duration in hours
 * @param {string} bucketName - The bucket name
 * @returns {Promise<string>} The signed get URL
 */
export const generateSignedGetUrl = async (key: string, durationHrs: number, bucketName: string) => {
    try {
        const params: GetObjectCommandInput = {
            Bucket: bucketName,
            Key: key,
        };
        const command = new GetObjectCommand(params);

        // Get a signed URL for the uploaded object (valid for 1 hour)
        return await getSignedUrl(s3, command, { expiresIn: 3600 });
    } catch (e) {
        log.error("Error encountered on server. Message:'{0}' when writing an object", { err: e });
    }
    return "";
};

/**
 * Delete an object from an S3 bucket.
 * @param bucket - The name of the bucket.
 * @param key - The key of the object to delete.
 */
export const deleteS3Object = async (bucket: string, key: string) => {
    try {
        await s3.send(new DeleteObjectCommand({ Bucket: bucket, Key: key }));
    } catch (e) {
        log.error("Error encountered on server. Message:'{0}' when deleting an object", { err: e });
    }
};

/**
 * Delete  objects from an S3 bucket.
 * @param bucket - The name of the bucket.
 * @param keys - The keys of the object to delete.
 */
export const deleteS3Objects = async (bucket: string, keys: string[]) => {
    try {
        const objectsToDelete = keys.map((Key) => {return { Key };});
        await s3.send(new DeleteObjectsCommand({ Bucket: bucket,       
            Delete: { Objects: objectsToDelete },
        }));
    } catch (e) {
        log.error("Error encountered on server. Message:'{0}' when deleting an object", { err: e });
    }
};

/**
 * Generates a thumbnail for a given media file stored in an S3 bucket.
 * @param bucket - The name of the S3 bucket where the media file is stored.
 * @param key - The key of the media file in the S3 bucket.
 * @param convertedMediaBucket - The name of the S3 bucket where the converted media file will be stored.
 * @param thumbnailKey - The key of the thumbnail file in the converted media bucket.
 * @param filePath - The path to the media file.
 * @returns A Promise that resolves when the thumbnail is generated and uploaded successfully.
 */
export const generateThumbnail = async (
    bucket: string,
    key: string,
    convertedMediaBucket: string,
    thumbnailKey: string,
    filePath: string,
) => {
    try {
        const params: GetObjectCommandInput = {
            Bucket: bucket,
            Key: key,
        };
        const response = await s3.send(new GetObjectCommand(params));
        const readStream = response.Body as Readable;
        await generateVideoThumbnail(
            convertedMediaBucket,
            thumbnailKey,
            readStream,
            s3,
            filePath,
        );
    
        log.debug("Thumbnail uploaded successfully:");
    } catch (error) {
        log.error("error: ", error);
    }
};
