import { PinpointClient, SendMessagesCommand } from "@aws-sdk/client-pinpoint";
import _omit from "lodash/omit";

import log from "../log";

/**
 * Pinpoint client for sending SMS notifications.
 */
const pinpoint = new PinpointClient({
    region: process.env.AWS_REGION,
});

/**
 * Represents an SMS message.
 */
export interface SmsMessage {
    /**
     * The destination phone numbers for the SMS message.
     */
    destinationPhoneNumbers: string[];

    /**
     * The origination phone number for the SMS message.
     */
    originationPhoneNumber: string;

    /**
     * The ID of the Pinpoint application associated with the SMS message.
     */
    pinpointApplicationId: string;

    /**
     * The body of the SMS message.
     */
    body: string;
}

/**
 * Checks if a phone number is a test phone number.
 * @param {string} phoneNumber - The phone number to check.
 * @returns {boolean} - Returns true if the phone number is a test phone number, false otherwise.
 */
const isTestPhoneNumber = (phoneNumber: string) => phoneNumber.startsWith("+1555");

/**
 * Sends an SMS message using AWS Pinpoint.
 * @param {SmsMessage} message - The SMS message to send.
 * @returns {Promise<any>} - A promise that resolves with the response from AWS Pinpoint.
 */
export async function send(message: SmsMessage): Promise<any> {
    let hasTestPhoneNumbers = false;
    const addressConfig = message.destinationPhoneNumbers.reduce(
        (map: any, phoneNumber: string): any => {
            /* eslint no-param-reassign: 0 */
            if (!isTestPhoneNumber(phoneNumber)) {
                map[phoneNumber] = {
                    ChannelType: "SMS",
                };
            } else {
                hasTestPhoneNumbers = true;
            }
            return map;
        }, {},
    );
    const messageData = {
        ApplicationId: message.pinpointApplicationId,
        MessageRequest: {
            Addresses: addressConfig,
            MessageConfiguration: {
                SMSMessage: {
                    Body: message.body,
                    MessageType: "TRANSACTIONAL",
                    OriginationNumber: message.originationPhoneNumber,
                },
            },
        },
    };
    if (hasTestPhoneNumbers && !Object.keys(messageData.MessageRequest.Addresses!).length) {
        log.info("Only test phone numbers included in request, so SMS messages will not be sent",
            { messageData, destinationPhoneNumbers: message.destinationPhoneNumbers });
        return { messageData };
    }
    const logContext = _omit(messageData, "MessageRequest.MessageConfiguration.SMSMessage.Body"); // Don't log body for sensitive messages (e.g. login codes)
    log.debug("Sending Pinpoint SMS message...", { message: logContext });
    try {
        const command = new SendMessagesCommand(messageData);
        const response = await pinpoint.send(command);
        // TODO: check for error code in statusCode and throw if doesn't equal success
        log.debug("Sent Pinpoint SMS message", response);
        return { response, messageData };
    } catch (error: any) {
        log.error("Error Sending Pinpoint SMS message to endpoint", { message: logContext }, error);
        throw error;
    }
}
