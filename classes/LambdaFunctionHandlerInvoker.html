import { InvokeCommand, LambdaClient } from "@aws-sdk/client-lambda";
import { Context, Handler } from "aws-lambda";

import { InvocationMode } from "./types";

/**
 * Represents the configuration for a Lambda function handler invoker.
 *
 * @template TEvent The type of the event object.
 * @template TResult The type of the result object.
 */
export interface LambdaFunctionHandlerInvokerConfig<TEvent = any, TResult = any> {
    /**
     * The invocation mode for the Lambda function handler.
     */
    invocationMode?: InvocationMode;

    /**
     * The handler function for the Lambda function.
     */
    handler: Handler<TEvent, TResult>;

    /**
     * The AWS region where the Lambda function is deployed.
     */
    awsRegion: string;

    /**
     * The name of the Lambda function.
     */
    lambdaFunctionName: string;
}

/**
 * Class for invoking Lambda function handlers either locally or remotely.
 *  - locally: by invoking the handler function passed into the constructor;
 *  - remotely: by using the AWS SDK to remotely invoke a deployed Lambda function.
 * @template TEvent - The type of the event payload.
 * @template TResult - The type of the result returned by the Lambda function.
 */
export class LambdaFunctionHandlerInvoker<TEvent = any, TResult = any> {

    /**
     * Valid invocation modes.
     * @type {InvocationMode[]}
     */
    static validModes = Object.values(InvocationMode);

    /**
     * The Lambda function handler.
     * @type {Handler | undefined}
     */
    readonly handler?: Handler;

    /**
     * The AWS Lambda client used for remote invocations.
     * @type {LambdaClient | undefined}
     */
    private readonly lambdaClient?: LambdaClient;

    /**
     * The invocation mode for the invoker.
     * @type {InvocationMode}
     */
    private readonly invocationMode: InvocationMode;

    /**
     * Constructs a new LambdaFunctionHandlerInvoker.
     * @param {LambdaFunctionHandlerInvokerConfig<TEvent, TResult>} config - Configuration options for the invoker.
     * @throws {Error} Throws an error if required configuration options are not provided.
     */
    constructor(private readonly config: LambdaFunctionHandlerInvokerConfig<TEvent, TResult>) {
        // Determine invocation mode based on config or environment variables
        if (config.invocationMode) {
            this.invocationMode = config.invocationMode;
        } else if (process.env.TEST_MODE) {
            this.invocationMode = process.env.TEST_MODE as InvocationMode;
        } else {
            this.invocationMode = InvocationMode.LOCAL_HANDLER;
        }
        
        // Check required configurations based on invocation mode
        if (this.invocationMode === InvocationMode.REMOTE_TRIGGER && !config.lambdaFunctionName) {
            throw new Error("'lambdaFunctionName' must be specified in options.");
        }
        if (this.invocationMode === InvocationMode.LOCAL_HANDLER && !config.handler) {
            throw new Error("'handler' must be specified in options.");
        }

        // Assign handler and initialize Lambda client for remote invocations
        this.handler = this.config.handler;

        if (this.invocationMode === InvocationMode.REMOTE_TRIGGER) {
            this.lambdaClient = new LambdaClient({
                region: config.awsRegion,
            });
        }
    }

    /**
     * Invokes a remote Lambda function with the given event payload.
     * @param event - The event payload to be passed to the Lambda function.
     * @returns A promise that resolves to the result of the Lambda function invocation.
     * @throws An error if the Lambda function returns a FunctionError.
     */
    async invokeRemote(event: TEvent) {
        const invokeCommand = new InvokeCommand({
            FunctionName: this.config.lambdaFunctionName,
            InvocationType: "RequestResponse",
            Payload: JSON.stringify(event),
        });
        const result = await this.lambdaClient!.send(invokeCommand);
        if (result.FunctionError) {
            throw new Error(`FunctionError returned from ${this.config.lambdaFunctionName}. Error details: ${result.Payload}`);
        }
        const decoder = new TextDecoder("utf-8");
        const payload = result.Payload ? decoder.decode(result.Payload) : "{}";
        return JSON.parse(payload) as TResult;
    }

    /**
     * Invokes the Lambda function locally with the given event.
     * @param event - The event object to pass to the Lambda function.
     * @returns A Promise that resolves to the result of the Lambda function.
     */
    async invokeLocal(event: TEvent) {
        const response = await this.handler!(event, {} as any as Context, () => { });
        return response as TResult;
    }

    /**
     * Invokes the Lambda function handler.
     * 
     * @param event - The event object to pass to the handler.
     * @returns A promise that resolves with the result of the invocation.
     */
    async invoke(event: TEvent) {
        return this.invocationMode === InvocationMode.REMOTE_TRIGGER
            ? this.invokeRemote(event) : this.invokeLocal(event);
    }
}
