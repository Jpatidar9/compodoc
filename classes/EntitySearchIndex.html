import { SearchOptions, Settings } from "@algolia/client-search";
import log from "@common-utils/log";
import algoliasearch, { SearchClient, SearchIndex } from "algoliasearch";

import { SearchableItem, SearchConfiguration } from "./search-types";

/**
 * Returns the full index name by concatenating the index name with an optional prefix.
 * @param indexName - The name of the index.
 * @param prefix - An optional prefix to prepend to the index name.
 * @returns The full index name.
 */
export const getFullIndexName = (indexName: string, prefix?: string) => `${prefix || ""}_${indexName}`;

/**
 * Light wrapper around the `SearchIndex` within the Algolia SDK.
 * Represents an entity search index.
 */
/**
 * Represents a search index for entities.
 */
export class EntitySearchIndex {
    /**
     * The search client used for interacting with the Algolia search service.
     */
    private readonly searchClient: SearchClient;

    /**
     * Represents the full index name used for searching entities.
     */
    private readonly fullIndexName: string = ""; // Assign an initial value to the fullIndexName property.

    /**
     * Represents the search index used for entity search.
     */
    private readonly index: SearchIndex = null!; // Initialize the index property.

    /**
     * The API key used for authentication with the Algolia search service.
     */
    private readonly apiKey: string;

    /**
     * Use async static ctor to allow for any settings (e.g. searchableAttributes) to be applied.
     * @param config - The search configuration.
     * @param indexName - The name of the index.
     */
    static async load(config: SearchConfiguration, indexName?: string, settings?: Settings) {
        const i = new EntitySearchIndex(config, indexName);
        if (settings && config.isWriteAccess) {
            const response = await i.index.setSettings(settings);
            await i.waitOnTasks([response.taskID]);
        }
        return i;
    }

    /**
     * Constructs a new instance of the EntitySearchIndex class.
     * @param config - The search configuration.
     * @param indexName - The name of the index.
     * @throws Error if `apiKey` or `applicationId` is not provided in the configuration.
     */
    private constructor(private readonly config: SearchConfiguration, indexName?: string) {
        if (!config.algolia.apiKey || !config.algolia.applicationId) {
            throw new Error("`apiKey` and `applicationId` must be provided");
        }
        this.searchClient = algoliasearch(config.algolia.applicationId, config.algolia.apiKey);
        this.apiKey = config.algolia.apiKey;

        if (indexName) {
            this.fullIndexName = getFullIndexName(indexName, config.indexPrefix);
            this.index = this.searchClient.initIndex(this.fullIndexName);
        }
    }

    /**
     * Generates a secured API key.
     * @returns A promise that resolves to the secured API key.
     */
    async generateSecuredApiKey() {
        const validUntil = Math.floor(Date.now() / 1000) + 3600;
        return this.searchClient.generateSecuredApiKey(this.apiKey, {
            validUntil,
            // restrictIndices: [this.fullIndexName],
        });
    }

    /**
     * Searches for items in the Algolia index based on the provided query.
     * @param query - The search query.
     * @param options - Optional search options.
     * @returns A promise that resolves to the search result.
     */
    async search(query: string, options?: SearchOptions) {
        const result = await this.index.search<SearchableItem>(query, options);
        return result;
    }

    /**
     * Saves objects to the index.
     * @param objects - The objects to be saved.
     * @param waitOnTasks - Indicates whether to wait for tasks to complete.
     * @returns A promise that resolves to the save result.
     */
    async putObjects(objects: SearchableItem[], waitOnTasks = false) {
        if (!objects.length) {
            return null;
        }
        const result = await this.index.saveObjects(objects);
        log.debug("Saved objects", { objects, result });
        if (waitOnTasks) {
            await this.waitOnTasks(result.taskIDs);
        }
        return result;
    }

    /**
     * Waits for tasks to complete.
     * @param taskIds - The IDs of the tasks to wait for.
     * @returns A promise that resolves when all tasks are completed.
     */
    async waitOnTasks(taskIds: number[]) {
        // Since writes are eventually consistent, we can use the waitTask whenever we want to wait before querying (e.g. in tests)
        await Promise.all(taskIds.map(async (taskId) => {
            await this.index.waitTask(taskId);
        }));
    }

    /**
     * Deletes an object from the index.
     * @param objectId - The ID of the object to delete.
     * @returns A promise that resolves to the delete result.
     */
    async deleteObject(objectId: string) {
        const result = await this.index.deleteObject(objectId);
        log.debug("Deleted object", { objectId, result });
        return result;
    }

    /**
     * Deletes multiple objects from the index.
     * @param objectIds - The IDs of the objects to delete.
     * @param waitOnTasks - Indicates whether to wait for tasks to complete.
     * @returns A promise that resolves to the delete result.
     */
    async deleteObjects(objectIds: string[], waitOnTasks = false) {
        if (!objectIds.length) {
            return null;
        }
        const result = await this.index.deleteObjects(objectIds);
        log.debug("Deleted objects", { objectIds, result });
        if (waitOnTasks) {
            await this.waitOnTasks(result.taskIDs);
        }
        return result;
    }
}
